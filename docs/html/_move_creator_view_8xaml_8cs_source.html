40 private void Chart_MouseDown(object sender, System.Windows.Input.MouseButtonEventArgs e)
41 {
42 var chart = (CartesianChart)FindName("chart");
43 var viewModel = (MoveCreatorViewModel)DataContext;
44
45 // gets the point in the UI coordinates.
46 var p = e.GetPosition(chart);
47
48 // scales the UI coordinates to the corresponding data in the chart.
49 // ScaleUIPoint returns an array of double
50 var scaledPoint = chart.ScaleUIPoint(new LvcPoint((float)p.X, (float)p.Y));
51 // where the X coordinate is in the first position
52 var x = scaledPoint[0];
53 // and the Y coordinate in the second position
54 var y = scaledPoint[1];
55 y = StepperMotorDriver.Instance.QuantitizeVelocity(y);
56 if (MVGTarget.Instance._creator_EditMode == false)
57 {
58 double? last_x = viewModel.Data.Last().X;
59 if (last_x != null)
60 {
61 if (x <= last_x || Math.Abs(x - (double)last_x) <= StepperMotorDriver.Instance.SmallestDelay)
62 {
63 x = (double)last_x;
64 x += StepperMotorDriver.Instance.SmallestDelay;
65 }
66 }
67
68 // finally add the new point to the data in our chart.
69 var ClickedPoint = new ObservablePoint(x, y);
70 viewModel.Data.Add(ClickedPoint);
71 MVGTarget.Instance.NotifyDataModified("+", ClickedPoint);
72 MVGTarget.Instance.CurrentMove.X.Max = x;
73 MVGTarget.Instance.NotifyFocusChanged(new ObservablePoint(x-2,x+5));
74 }
75 else
76 {
77 for(int i = 0; i < viewModel.Data.Count; i++)
78 {
79 if (viewModel.Data[i].X > x)
80 {
81 if ( i>0)
82 {
83 if ((x - (double)viewModel.Data[i - 1].X) < ((double)viewModel.Data[i].X - x) && i>1)
84 {
85 viewModel.Data[i - 1].X = x;
86 viewModel.Data[i-1].Y = y;
87 }
88 else
89 {
90 viewModel.Data[i].X = x;
91 viewModel.Data[i].Y = y;
92 }
93 }
94 break;
95 }
96 else if (i == viewModel.Data.Count - 1)
97 {
98 viewModel.Data[i].X = x;
99 viewModel.Data[i].Y = y;
100 }
101 }
102 }
